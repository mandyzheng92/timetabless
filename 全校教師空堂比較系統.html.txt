        // 整理時間表內容：只提取班別與科目。例如「6B 中文 陳 604」或「6B 中文」-> 變為「6B 中文」
        function cleanSubjectText(text) {
            if (!text) return "";
            let cleaned = text.replace(/小息Recess|午膳 Lunch|午息Recess|班主任課\/早讀|小息/g, "").trim();
            if (!cleaned) return text;

            // CLP 備課處理：改為年級+備課，例如「CLP029 P5文 珊,伍,梁」簡化為「P5 備課」
            if (cleaned.includes('CLP')) {
                const match = cleaned.match(/([pP]?)([1-6])/);
                if (match) {
                    return "P" + match[2] + " 備課";
                }
                return "協同備課";
            }

            // 分割字串。通常是 [班別, 科目, 人名, 教室]
            const parts = cleaned.split(/\s+/);
            if (parts.length >= 2) {
                // 回傳前兩個元素，即「班別 + 科目」
                return parts[0] + " " + parts[1];
            }

            // 處理括號類型如 "CLP035 P5科 (鳳,炎,梁)" -> "P5 備課" (符合CLP改動要求)
            if (cleaned.includes('(')) {
                const match = cleaned.match(/([pP]?)([1-6])/);
                if (match) {
                    return "P" + match[2] + " 備課";
                }
                const pureText = cleaned.split('(')[0].trim();
                const pureParts = pureText.split(/\s+/);
                if (pureParts.length >= 2) {
                    return pureParts[0] + " " + pureParts[1];
                }
                return pureText;
            }

            return cleaned;
        }

        function isEmptyPeriod(text) {
            if (!text) return true;
            const cleaned = text.trim();
            return cleaned === "" || cleaned === "　" || cleaned === "空堂" || cleaned === "課前集會" || cleaned === "—";
        }

        // 統一教師姓名格式為：「全名 + 職稱」（普通老師使用「老師」而不是「教師」）
        function getFullDisplayName(name, title) {
            if (!title || title === "老師" || title === "教師") {
                return name + "老師";
            }
            return name + title;
        }

        // 將排課資料轉換為 MORNING_SLOTS 格式
        function convertTeacherScheduleToMorning(scheduleArray) {
            const morning = createEmptySchedule();
            scheduleArray.forEach(item => {
                const dayKey = item.day;
                const timeSlot = MORNING_SLOTS.findIndex(s => s.time === item.time);
                if (timeSlot !== -1 && morning[dayKey]) {
                    morning[dayKey][timeSlot] = item.detail;
                }
            });
            return morning;
        }

        function switchTab(tabId) {
            // 隱藏所有 section
            document.querySelectorAll('main > section').forEach(s => s.classList.add('hidden'));
            document.getElementById(tabId + '-content').classList.remove('hidden');

            // 更新按鈕樣式
            const tabs = ['admin-tab', 'view-tab', 'compare-tab'];
            tabs.forEach(id => {
                const btn = document.getElementById('btn-' + id);
                if (id === tabId) {
                    btn.classList.add('bg-rose-500', 'shadow-inner');
                    btn.classList.remove('bg-rose-300');
                } else {
                    btn.classList.remove('bg-rose-500', 'shadow-inner');
                    btn.classList.add('bg-rose-300');
                }
            });

            // 進入 subdivision 頁面時的資料重整
            if (tabId === 'compare-tab') {
                renderTeacherPills();
                runComparison();
            } else if (tabId === 'view-tab') {
                populateTeacherSelect();
                renderIndividualSchedule();
            } else if (tabId === 'admin-tab') {
                populateAdminTeacherSelect();
                handleAdminTeacherSelectChange();
            }
        }

        function showMessage(title, content, showCancel = false, confirmCallback = null) {
            document.getElementById('msg-title').innerText = title;
            document.getElementById('msg-content').innerText = content;
            
            const actions = document.getElementById('msg-actions');
            actions.innerHTML = '';
            
            if (showCancel) {
                const cancelBtn = document.createElement('button');
                cancelBtn.innerText = '取消';
                cancelBtn.className = 'bg-gray-100 hover:bg-gray-200 text-gray-700 px-4 py-2 rounded-lg font-medium transition';
                cancelBtn.onclick = closeMsgBox;
                actions.appendChild(cancelBtn);
            }
            
            const confirmBtn = document.createElement('button');
            confirmBtn.innerText = '確定';
            confirmBtn.className = 'bg-rose-500 hover:bg-rose-600 text-white px-4 py-2 rounded-lg font-medium transition shadow';
            confirmBtn.onclick = () => {
                closeMsgBox();
                if (confirmCallback) confirmCallback();
            };
            actions.appendChild(confirmBtn);

            const modal = document.getElementById('message-box');
            modal.classList.remove('hidden');
            // Animate in
            setTimeout(() => {
                modal.querySelector('div').classList.remove('scale-95');
                modal.querySelector('div').classList.add('scale-100');
            }, 10);
        }

        function closeMsgBox() {
            const modal = document.getElementById('message-box');
            modal.querySelector('div').classList.remove('scale-100');
            modal.querySelector('div').classList.add('scale-95');
            setTimeout(() => {
                modal.classList.add('hidden');
            }, 200);
        }


        // ==================== 1. 個人時間表檢視邏輯 ====================

        function populateTeacherSelect() {
            const select = document.getElementById('teacher-select');
            const currentValue = select.value;
            select.innerHTML = '';
            
            const sortedNames = Object.keys(TEACHERS_DATABASE).sort((a,b) => {
                const order = {"副校長": 1, "助理校長": 2, "主任": 3, "老師": 4, "教師": 4};
                const weightA = order[TEACHERS_DATABASE[a].title] || 9;
                const weightB = order[TEACHERS_DATABASE[b].title] || 9;
                if (weightA !== weightB) return weightA - weightB;
                return a.localeCompare(b, 'zh-Hant');
            });

            sortedNames.forEach(name => {
                const opt = document.createElement('option');
                opt.value = name;
                // 統一格式：全名 + 職稱
                opt.textContent = getFullDisplayName(name, TEACHERS_DATABASE[name].title);
                select.appendChild(opt);
            });
            
            if (currentValue && TEACHERS_DATABASE[currentValue]) {
                select.value = currentValue;
            }
        }

        function renderIndividualSchedule() {
            const select = document.getElementById('teacher-select');
            const teacherName = select.value;
            const searchInput = document.getElementById('search-course').value.trim().toLowerCase();
            const tbody = document.getElementById('morning-table-body');
            tbody.innerHTML = '';

            if (!teacherName || !TEACHERS_DATABASE[teacherName]) return;

            const schedule = TEACHERS_DATABASE[teacherName].morning;

            MORNING_SLOTS.forEach((slot, index) => {
                const tr = document.createElement('tr');
                if (slot.isFixed) tr.classList.add('bg-gray-50');

                // 時間 Slot 與標題（「班主任課/早讀」、「小息」、「午膳」、「午息」）變黑色字
                let labelClass = slot.isFixed ? "text-black font-semibold" : "text-gray-500 font-normal";
                let html = `<td class="p-3 border-r border-rose-100 font-medium whitespace-nowrap bg-white sticky left-0 shadow-[2px_0_5px_-2px_rgba(0,0,0,0.1)]">
                                <span class="text-black font-semibold">${slot.time}</span><br>
                                <span class="text-[11px] ${labelClass}">${slot.label}</span>
                            </td>`;

                for (let i = 1; i <= 6; i++) {
                    const dayKey = `DAY ${i}`;
                    const rawDetail = schedule[dayKey][index] || "";
                    // 經過 cleanSubjectText 整理成「班別 + 科目」
                    const detail = slot.isFixed ? rawDetail : cleanSubjectText(rawDetail);
                    let isMatch = searchInput && detail.toLowerCase().includes(searchInput);
                    let cellClass = "p-3 border-r border-rose-50 transition-colors";

                    if (slot.isFixed) {
                         // 固定時段/休息時間在表格內也變為黑色字
                         cellClass += " text-black font-semibold text-xs";
                    } else if (isMatch) {
                        cellClass += " bg-yellow-100 font-bold text-yellow-800 shadow-inner border-y border-yellow-200";
                    } else if (isEmptyPeriod(detail)) {
                        // 空堂使用粉紅色標示且字體大小與上課的相同 (.free-period-cell text-sm)
                        cellClass += " free-period-cell text-sm shadow-inner";
                    } else {
                        // 上課空格強制用白色背景
                        cellClass += " text-gray-700 bg-white text-sm";
                    }

                    const displayDetail = slot.isFixed ? detail : (isEmptyPeriod(detail) ? '空堂' : detail);
                    html += `<td class="${cellClass}">${displayDetail}</td>`;
                }
                tr.innerHTML = html;
                tbody.appendChild(tr);
            });
        }

        function resetIndividualSearch() {
            document.getElementById('search-course').value = '';
            renderIndividualSchedule();
        }

        function handleAdminTeacherSelectChange() {
            const select = document.getElementById('edit-teacher-select');
            const container = document.getElementById('new-teacher-input-container');
            const nameInput = document.getElementById('edit-teacher-name');
            const titleSelect = document.getElementById('edit-teacher-title');
            
            if (select.value === '__NEW__') {
                container.classList.remove('hidden');

                nameInput.value = select.value;
                titleSelect.value = TEACHERS_DATABASE[select.value].title || '老師';
            }
            renderEditSlots();
        }

            MORNING_SLOTS.forEach((slot, index) => {
                const div = document.createElement('div');
                div.className = 'flex items-center gap-3 p-2 hover:bg-rose-50 rounded-lg transition';
                
                if (slot.isFixed) {
                    div.innerHTML = `
                        <label class="w-24 text-sm text-black font-semibold text-right pr-2">${slot.label}</label>
                        <input type="text" disabled value="${slot.label}" class="flex-1 p-2 bg-gray-100 border border-gray-200 rounded-lg text-sm text-black font-semibold cursor-not-allowed">
                    `;
                } else {
                    div.innerHTML = `
                        <label class="w-24 text-sm font-semibold text-gray-700 text-right pr-2">${slot.label}</label>
                        <input type="text" id="edit-slot-${index}" value="${schedule[day][index] || ''}"
                               class="flex-1 p-2 bg-white border border-rose-200 rounded-lg text-sm focus:ring-2 focus:ring-rose-400 focus:border-rose-400 transition"
                               placeholder="留空即表示空堂">
                    `;
                }
                container.appendChild(div);
            });
        }

        function saveTeacherData() {
            const select = document.getElementById('edit-teacher-select');
            const isNew = select.value === '__NEW__';
            const name = isNew ? document.getElementById('edit-teacher-name').value.trim() : select.value;
            
            if (!name) { 
                showMessage("錯誤", "請先輸入教師姓名");
                return; 
            }
            const title = document.getElementById('edit-teacher-title').value;
            const day = document.getElementById('edit-day-select').value;

            if (!TEACHERS_DATABASE[name]) {
                TEACHERS_DATABASE[name] = { title: title, morning: createEmptySchedule() };
            } else {
                TEACHERS_DATABASE[name].title = title;
            }

            MORNING_SLOTS.forEach((slot, index) => {
                if (!slot.isFixed) {
                    const input = document.getElementById(`edit-slot-${index}`);
                    if (input) {
                        TEACHERS_DATABASE[name].morning[day][index] = input.value.trim();
                    }
                }
            });

            showMessage("儲存成功", `已更新 ${name} 的 ${day} 課表資料！`);
            
            // 重新整理頁面的選單與名單
            populateTeacherSelect();
            populateAdminTeacherSelect();
            
            // 自動選擇剛才儲存的老師
            document.getElementById('edit-teacher-select').value = name;
            handleAdminTeacherSelectChange();
            
            renderTeacherPills();
            runComparison();
        }

        function confirmDeleteTeacher() {
            const select = document.getElementById('edit-teacher-select');
            const isNew = select.value === '__NEW__';
            const name = isNew ? document.getElementById('edit-teacher-name').value.trim() : select.value;
            
            if (!name || !TEACHERS_DATABASE[name]) {
                showMessage("錯誤", "找不到該教師資料，無法刪除");
                return;
            }

            showMessage("確認刪除", `確定要刪除 ${name} 的所有資料嗎？此操作無法還原。`, true, () => {
                delete TEACHERS_DATABASE[name];
                selectedTeachers.delete(name);
                
                populateTeacherSelect();
                populateAdminTeacherSelect();
                
                // 重設回新增畫面
                document.getElementById('edit-teacher-select').value = '__NEW__';
                handleAdminTeacherSelectChange();
                
                renderTeacherPills();
                runComparison();
                
                showMessage("刪除成功", `已刪除 ${name} 的資料`);
            });
        }

        // ==================== 3. 多人空堂比對邏輯 ====================

        function renderTeacherPills() {
            const container = document.getElementById('teacher-pills-container');
            const searchText = document.getElementById('teacher-search').value.toLowerCase();
            container.innerHTML = '';

            const sortedNames = Object.keys(TEACHERS_DATABASE).sort((a,b) => {
                const order = {"副校長": 1, "助理校長": 2, "主任": 3, "老師": 4, "教師": 4};
                return (order[TEACHERS_DATABASE[a].title] || 9) - (order[TEACHERS_DATABASE[b].title] || 9);
            });

            sortedNames.forEach(name => {
                const title = TEACHERS_DATABASE[name].title;
                const formattedFullName = getFullDisplayName(name, title);
                
                if (searchText && !formattedFullName.toLowerCase().includes(searchText)) {
                    return;
                }

                const btn = document.createElement('button');
                const isSelected = selectedTeachers.has(name);
                
                btn.className = `pill-button ${isSelected ? 'active' : 'inactive'}`;

                // 移除 PILL 選項中的任何表情符號 (Emoji)，僅保留文字名稱與職稱
                btn.innerHTML = `${formattedFullName}`;
                btn.onclick = () => {
                    if (selectedTeachers.has(name)) {
                        selectedTeachers.delete(name);
                    } else {
                        selectedTeachers.add(name);
                    }
                    renderTeacherPills();
                    runComparison();
                };
                container.appendChild(btn);
            });
        }

        function selectCoreTeam(type) {
            // 點選「所有老師」或任一職級篩選時，可直接轉其他選項（即先清空再重新加入對應對象）
            selectedTeachers.clear();
            if (type === '所有老師') {
                Object.keys(TEACHERS_DATABASE).forEach(t => selectedTeachers.add(t));
            } else {
                Object.keys(TEACHERS_DATABASE).forEach(t => {
                    const title = TEACHERS_DATABASE[t].title;
                    // 將 "老師" 按鈕支援選取職稱為 "老師" 與 "教師" 的對象
                    if (title === type || (type === '老師' && (title === '老師' || title === '教師'))) {
                        selectedTeachers.add(t);
                    }
                });
            }
            renderTeacherPills();
            runComparison();
        }

        function toggleAllTeachers(forceClear = false) {
             if (forceClear) {
                 selectedTeachers.clear();
             } else {
                 if (selectedTeachers.size === Object.keys(TEACHERS_DATABASE).length) {
                     selectedTeachers.clear();
                 } else {
                     Object.keys(TEACHERS_DATABASE).forEach(t => selectedTeachers.add(t));
                 }
             }
             renderTeacherPills();
             runComparison();
        }

        function runComparison() {
            const day = document.getElementById('compare-day-select').value;
            document.getElementById('matrix-day-badge').innerText = day;
            
            const thead = document.getElementById('comparison-table-head');
            const tbody = document.getElementById('comparison-table-body');
            const summaryTitle = document.getElementById('summary-title');
            const summaryDesc = document.getElementById('summary-desc');

            thead.innerHTML = '';
            tbody.innerHTML = '';

            const teachers = Array.from(selectedTeachers);
            if (teachers.length === 0) {
                summaryTitle.textContent = "請於左側選擇至少一位教師";
                summaryDesc.textContent = "點擊教師名稱，即可檢視並對比他們的課表";
                tbody.innerHTML = `<tr><td class="p-8 text-center text-gray-400" colspan="2">尚未選擇教師</td></tr>`;
                return;
            }

            summaryTitle.textContent = `已選 ${teachers.length} 位教師 (${day})`;
            summaryDesc.textContent = "所有被選擇教師均顯示「空堂」的時段，將會整行以粉紅色高亮顯示（完美會議時段）。";

            // 建立表頭
            let headerHtml = `<tr>
                <th class="p-3 w-32 border-r border-rose-200 bg-[#FDF8F8] sticky left-0 z-10 shadow-[2px_0_5px_-2px_rgba(0,0,0,0.05)] text-black font-semibold">時間 Slot</th>`;
            teachers.forEach(t => {
                // 統一格式：全名 + 職稱
                const formattedName = getFullDisplayName(t, TEACHERS_DATABASE[t].title);
                headerHtml += `<th class="p-3 text-center border-r border-rose-200 min-w-[120px]">
                                <div class="font-bold text-gray-800">${formattedName}</div>
                               </th>`;
            });
            headerHtml += `</tr>`;
            thead.innerHTML = headerHtml;

            // 建立表身
            let matchCount = 0;

            MORNING_SLOTS.forEach((slot, index) => {
                const tr = document.createElement('tr');
                let isPerfectMatch = true;
                
                // 第一列 (時間)，固定時段也改為黑色字
                let labelClass = slot.isFixed ? "text-black font-semibold" : "text-gray-500 font-normal";
                let rowHtml = `<td class="p-3 border-r border-rose-100 font-medium whitespace-nowrap bg-white sticky left-0 z-10 shadow-[2px_0_5px_-2px_rgba(0,0,0,0.05)]">
                                <span class="text-black font-semibold">${slot.time}</span><br>
                                <span class="text-[11px] ${labelClass}">${slot.label}</span>
                               </td>`;

                // 教師列
                let rowDetails = [];
                teachers.forEach(t => {
                    const rawDetail = TEACHERS_DATABASE[t].morning[day][index] || "";
                    const detail = slot.isFixed ? rawDetail : cleanSubjectText(rawDetail);
                    const isEmpty = isEmptyPeriod(detail);
                    
                    if (!isEmpty) isPerfectMatch = false;

                    if (slot.isFixed) {
                        // 固定時段在多人比對表中也統一顯示為黑色粗體字且大小與一般一樣
                        rowDetails.push(`<td class="p-3 border-r border-rose-100 bg-gray-50 text-black font-semibold text-sm">${detail || '-'}</td>`);
                    } else if (isEmpty) {
                        // 空堂高亮塗粉紅色且字體大小與上課的相同 (.free-period-cell text-sm)
                        rowDetails.push(`<td class="p-3 border-r border-rose-100 free-period-cell text-sm shadow-inner">空堂</td>`);
                    } else {
                        // 一般上課格子字體大小相同 (text-sm)
                        rowDetails.push(`<td class="p-3 border-r border-rose-100 text-gray-700 text-sm bg-white">${detail}</td>`);
                    }
                });

                if (isPerfectMatch && !slot.isFixed && teachers.length > 1) {
                    tr.classList.add('perfect-match-row');
                    matchCount++;
                }

                tr.innerHTML = rowHtml + rowDetails.join('');
                tbody.appendChild(tr);
            });

            // 更新總結文字
            if (teachers.length > 1) {
                if (matchCount > 0) {
                    summaryDesc.innerHTML = `<strong class="text-rose-600">找到 ${matchCount} 個完美會議時段！</strong>已於下方表格粉紅色標記。`;
                } else {
                    summaryDesc.innerHTML = `<span class="text-gray-500">很可惜，${day} 這 ${teachers.length} 位教師沒有共同的空堂。</span>`;
                }
            }
        }
    </script>
